// VERVANG DIT DOOR JOUW EIGEN CLOUDFLARE WORKER URL (ZONDER / ER ACHTER!) const CLOUDFLARE_PROXY_URL = "https://voetbal-api-proxy.jouwnaam.workers.dev"; async function startScorebordsysteem() { const container = document.getElementById('matches-container'); try { // We checken de status via Cloudflare let response = await fetch(`${CLOUDFLARE_PROXY_URL}/status`); let data = await response.json(); if (data.errors && Object.keys(data.errors).length > 0) { const foutType = Object.keys(data.errors)[0]; const foutBericht = data.errors[foutType]; container.innerHTML = `
API Fout (${foutType}):
${foutBericht}
`; return; } if (data && data.response) { const sub = data.response.subscription; if (sub.active !== true) { container.innerHTML = '
API Abonnement inactief.
'; return; } haalLaatsteResultatenOp(); } } catch (e) { haalLaatsteResultatenOp(); } } async function haalLaatsteResultatenOp() { const container = document.getElementById('matches-container'); try { // We halen de uitslagen op via Cloudflare (league=39 voor Premier League) let response = await fetch(`${CLOUDFLARE_PROXY_URL}/fixtures?league=39&status=FT&last=3`); let data = await response.json(); if (data.errors && Object.keys(data.errors).length > 0) { const foutType = Object.keys(data.errors)[0]; const foutBericht = data.errors[foutType]; container.innerHTML = `
Data Fout (${foutType}):
${foutBericht}
`; return; } container.innerHTML = ''; if (data && data.response && data.response.length > 0) { data.response.forEach(item => { const row = document.createElement('div'); row.className = 'match-row'; const teams = item.teams; const goals = item.goals; row.innerHTML = `
${teams.home.name}
${goals.home ?? 0} - ${goals.away ?? 0}
${teams.away.name}
Afgelopen
`; container.appendChild(row); }); } else { container.innerHTML = '
Geen uitslagen gevonden.
'; } } catch (err) { container.innerHTML = '
Verbindingsfout via Cloudflare.
'; } } // Start direct op bij laden startScorebordsysteem(); // Ververs elke 30 minuten (mag nu gerust vaker, Cloudflare vangt de klappen op!) setInterval(startScorebordsysteem, 1800000); // Live Klok function updateKlokEnDatum() { const nu = new Date(); document.getElementById('live-klok').textContent = `${String(nu.getHours()).padStart(2, '0')}:${String(nu.getMinutes()).padStart(2, '0')}`; const dagen = ['Zondag', 'Maandag', 'Dinsdag', 'Woensdag', 'Donderdag', 'Vrijdag', 'Zaterdag']; const maanden = ['januari', 'februari', 'maart', 'april', 'mei', 'juni', 'juli', 'augustus', 'september', 'oktober', 'november', 'december']; document.getElementById('live-datum').textContent = `${dagen[nu.getDay()]} ${nu.getDate()} ${maanden[nu.getMonth()]}`; } updateKlokEnDatum(); setInterval(updateKlokEnDatum, 1000); // Nieuwsbalk (NU.nl RSS) let nieuwsArtikelen = []; let huidigArtikelIndex = 0; async function haalNieuwsOp() { try { const response = await fetch(`https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent('https://www.nu.nl/rss/Algemeen')}`); const data = await response.json(); if (data.status === 'ok' && data.items.length > 0) { nieuwsArtikelen = data.items.map(item => item.title.replace(/\[\w+\]/gi, '').trim()); toonVolgendNieuws(); } } catch (e) {} } function toonVolgendNieuws() { if (nieuwsArtikelen.length === 0) return; document.getElementById('news-ticker').textContent = nieuwsArtikelen[huidigArtikelIndex]; pasTekstGrootteAan(); huidigArtikelIndex = (huidigArtikelIndex + 1) % nieuwsArtikelen.length; } function pasTekstGrootteAan() { const ticker = document.getElementById('news-ticker'); const container = document.getElementById('news-text-container'); let huidigeGrootte = parseFloat(window.getComputedStyle(container).getPropertyValue('font-size')); ticker.style.fontSize = ''; while ((ticker.scrollHeight > container.clientHeight || ticker.scrollWidth > container.clientWidth * 2) && huidigeGrootte > 8) { huidigeGrootte -= 0.5; ticker.style.fontSize = huidigeGrootte + "px"; } } haalNieuwsOp(); setInterval(toonVolgendNieuws, 8000); setInterval(haalNieuwsOp, 600000); window.addEventListener('resize', pasTekstGrootteAan);
